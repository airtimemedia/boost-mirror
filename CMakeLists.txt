function(at_boost_add_library LIB)
  set(_sources ${ARGN})
  set(_target ${LIB})

  add_library(${_target} STATIC ${_sources})
  target_include_directories(${_target} PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})
endfunction(at_boost_add_library)
